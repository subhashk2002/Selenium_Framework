Step 2: Use in Your Tests
--------
@Test
public void myTest() {
    // Click with automatic retry
    CommonActions.clickWithRetry(By.id("button"));

    // Clear and type with verification
    CommonActions.clearAndType(By.id("input"), "text");

    // Wait for element visibility
    CommonActions.waitForElementVisible(By.id("element"));

    // Fill and submit form
    Map<By, String> data = new HashMap<>();
    data.put(By.id("field1"), "value1");
    FormActions.fillAndSubmitForm(data, By.id("submitBtn"));
}
